<script>
// Initialize data storage
let mosqueData = JSON.parse(localStorage.getItem('mosqueData')) || [];

// Update letter template
function updateLetterTemplate() {
const senderName = document.getElementById('senderName').value || '[Your Name]';
const senderOrg = document.getElementById('senderOrg').value || '[Your Organization]';
document.getElementById('displaySenderName').textContent = senderName;
document.getElementById('displaySenderOrg').textContent = senderOrg;
}

// Copy letter to clipboard
function copyLetter() {
const letterContent = document.getElementById('letterTemplate').innerText;
navigator.clipboard.writeText(letterContent).then(() => {
alert('Letter copied to clipboard!');
});
}

// Add new mosque contact
function addMosqueContact() {
const name = document.getElementById('mosqueName').value;
const location = document.getElementById('mosqueLocation').value;
const email = document.getElementById('mosqueEmail').value;
const phone = document.getElementById('mosquePhone').value;
const contact = document.getElementById('mosqueContact').value;

if (!name || !location || !email) {
alert('Please fill in mosque name, location, and email address.');
return;
}

const mosque = {
id: Date.now(),
name,
location,
email,
phone,
contact,
status: 'Not Contacted',
dateContacted: '',
response: '',
notes: ''
};

mosqueData.push(mosque);
saveMosqueData();
updateMosqueTable();
updateStatistics();

// Clear form
document.getElementById('mosqueName').value = '';
document.getElementById('mosqueLocation').value = '';
document.getElementById('mosqueEmail').value = '';
document.getElementById('mosquePhone').value = '';
document.getElementById('mosqueContact').value = '';
}

// Update mosque status
function updateMosqueStatus(id, status, dateContacted = '') {
const mosque = mosqueData.find(m => m.id === id);
if (mosque) {
mosque.status = status;
if (dateContacted) mosque.dateContacted = dateContacted;
if (status === 'Contacted' && !mosque.dateContacted) {
mosque.dateContacted = new Date().toLocaleDateString();
}
saveMosqueData();
updateMosqueTable();
updateStatistics();
}
}

// Update mosque response
function updateMosqueResponse(id, response) {
const mosque = mosqueData.find(m => m.id === id);
if (mosque) {
mosque.response = response;
if (response && mosque.status === 'Contacted') {
mosque.status = 'Responded';
}
saveMosqueData();
updateMosqueTable();
updateStatistics();
}
}

// Delete mosque
function deleteMosque(id) {
if (confirm('Are you sure you want to delete this mosque contact?')) {
mosqueData = mosqueData.filter(m => m.id !== id);
saveMosqueData();
updateMosqueTable();
updateStatistics();
}
}

// Update mosque table
function updateMosqueTable() {
const tbody = document.getElementById('mosqueTableBody');
tbody.innerHTML = '';

mosqueData.forEach(mosque => {
const row = document.createElement('tr');
row.className = 'tracking-row';
const statusColor = {
'Not Contacted': 'gray',
'Contacted': 'yellow',
'Responded': 'green',
'No Response': 'red'
};

row.innerHTML = `
<td class="px-4 py-2 font-medium">${mosque.name}</td>
<td class="px-4 py-2">${mosque.location}</td>
<td class="px-4 py-2">
<div>${mosque.email}</div>
${mosque.phone ? `<div class="text-gray-500 text-xs">${mosque.phone}</div>` : ''}
${mosque.contact ? `<div class="text-gray-500 text-xs">${mosque.contact}</div>` : ''}
</td>
<td class="px-4 py-2">
<select onchange="updateMosqueStatus(${mosque.id}, this.value)"
class="text-xs border rounded px-2 py-1 bg-${statusColor[mosque.status]}-100 text-${statusColor[mosque.status]}-800">
<option value="Not Contacted" ${mosque.status === 'Not Contacted' ? 'selected' : ''}>Not Contacted</option>
<option value="Contacted" ${mosque.status === 'Contacted' ? 'selected' : ''}>Contacted</option>
<option value="Responded" ${mosque.status === 'Responded' ? 'selected' : ''}>Responded</option>
<option value="No Response" ${mosque.status === 'No Response' ? 'selected' : ''}>No Response</option>
</select>
</td>
<td class="px-4 py-2 text-xs">${mosque.dateContacted || '-'}</td>
<td class="px-4 py-2">
<input type="text" value="${mosque.response}"
onchange="updateMosqueResponse(${mosque.id}, this.value)"
placeholder="Response notes..."
class="text-xs border rounded px-2 py-1 w-full">
</td>
<td class="px-4 py-2">
<button onclick="deleteMosque(${mosque.id})"
class="text-red-600 hover:text-red-800 text-xs">
<i class="fas fa-trash"></i>
</button>
</td>
`;
tbody.appendChild(row);
});
}

// Update statistics
function updateStatistics() {
const total = mosqueData.length;
const contacted = mosqueData.filter(m => m.status === 'Contacted' || m.status === 'Responded' || m.status === 'No Response').length;
const responded = mosqueData.filter(m => m.status === 'Responded').length;
const responseRate = contacted > 0 ? Math.round((responded / contacted) * 100) : 0;

document.getElementById('totalMosques').textContent = total;
document.getElementById('contactedMosques').textContent = contacted;
document.getElementById('respondedMosques').textContent = responded;
document.getElementById('successRate').textContent = responseRate + '%';
}

// Save data to localStorage
function saveMosqueData() {
localStorage.setItem('mosqueData', JSON.stringify(mosqueData));
}

// Export data
function exportData() {
const dataStr = JSON.stringify(mosqueData, null, 2);
const dataBlob = new Blob([dataStr], {type: 'application/json'});
const url = URL.createObjectURL(dataBlob);
const link = document.createElement('a');
link.href = url;
link.download = 'mosque_campaign_data.json';
link.click();
}

// Import data
function importData(event) {
const file = event.target.files[0];
if (file) {
const reader = new FileReader();
reader.onload = function(e) {
try {
const importedData = JSON.parse(e.target.result);
if (Array.isArray(importedData)) {
mosqueData = importedData;
saveMosqueData();
updateMosqueTable();
updateStatistics();
alert('Data imported successfully!');
} else {
alert('Invalid file format');
}
} catch (error) {
alert('Error reading file');
}
};
reader.readAsText(file);
}
}

// Clear all data
function clearAllData() {
if (confirm('Are you sure you want to clear all data? This cannot be undone.')) {
mosqueData = [];
saveMosqueData();
updateMosqueTable();
updateStatistics();
}
}

// Initialize page
document.addEventListener('DOMContentLoaded', function() {
updateMosqueTable();
updateStatistics();
document.getElementById('lastUpdated').textContent = new Date().toLocaleDateString();
});
